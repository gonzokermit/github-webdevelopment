for line in file:

line = line.strip()

print("IP-Forward ist auf {} gesetzt!".format(line))

ans,unans = srp(Ether(dst="ff:ff:ff:ff:ff:ff")/ARP(pdst=ip),timeout=2,verbose=0)

if ans:
